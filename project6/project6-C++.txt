    return response;

void FillArray(int array[], int size){
    int temp = 0;
    for (int i=0;i<size; i++){
        temp = rand() % 11;
        array[i] = temp;
        std::cout<<array[i]<<" ";
    }
    std::cout<<std::endl;
}

void CalculateArraySum(int array[], int size){
    int sum = 0;
    for(int i=0;i<size;i++){
        sum += array[i];
    }
    std::cout << "The sum of the array is: "<<sum<<std::endl;
}
